pipeline {
    agent any

    stages {
        stage('Generate') {
            steps {
                checkout scm
            } 
        }    
        stage('Build') {
            steps {
                echo 'Build project'
                sh "chmod +x gradlew"
                sh "./gradlew build test --stacktrace" //run a gradle task
                step( [ $class: 'JacocoPublisher' ] )
                    }
            }      
        stage('Unit and Integration Test') {
            steps {
                echo 'Testing..'
                junit '**/test-results/test/*.xml'
            }
        }
        stage('Deploy') {
            steps {
                echo 'Deploying....'
            }
        }
    }
}
