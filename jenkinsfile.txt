pipeline {
    agent any

    stages {
        stage('Generate') {
            steps {
                checkout scm
            } 
        }    
        stage('Gradle Build') {
            steps {
                echo 'Build project'
                sh "chmod +x gradlew"
                sh "./gradlew build --stacktrace" //run a gradle task
                step( [ $class: 'JacocoPublisher' ] )
                    }
            }      
        stage('Unit and Integration Test') {
            steps {
                echo 'Testing..'
                //junit '**/test-results/test/*.xml'
                junit(
                    allowEmptyResults: true,
                    testResults: '**/test-reports/*.xml'
                ) 
            }
        }
        stage('NodeJS Build') {
            steps {
                sh 'npm install'

            }
        }
    }
}
